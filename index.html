%matplotlib inline
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_pickle("/workspace/labels/raw.pkl")

countdf = df.loc[:, ['numValidRegions', 'region0']].groupby('numValidRegions').count()
distr = countdf.to_dict()['region0']

plt.figure(figsize=(10, 5))
plt.barh(distr.keys(), distr.values(), align='center')
    
for charNb, frq in distr.items():
    plt.text(frq+5, charNb-.08, str(frq), fontweight='bold')

plt.xlabel('Number of license plates')
plt.ylabel('Number of characters')
plt.title('Frequency distribution of the license plates')
plt.ylim(min(distr.keys())-0.5, max(distr.keys())+0.5)
plt.show()
